    #!/bin/ash
    if [ "$1" == "m" ]; then
        echo "o" > /dev/ttyS1
        VIEWERS=`wget -q -O - "http://www.livestream.com/guide/search?count=&search_tag=relevantchurch" | grep -i "Viewers now:" | cut -f 2 -d : | cut -f 1 -d '"'`
        echo "d$VIEWERS" > /dev/ttyS1
    elif [ "$1" == "s" ]; then
        TIME=`date +%I%M`
        echo "s$TIME" > /dev/ttyS1
    elif [ "$1" == "o" ]; then
        echo "o" > /dev/ttyS1
    fi
